# Modal

## Basic usage

<Playground>
  <Modal.Wrapper>
    {(props) => (
      <Modal {...props} trigger={
        <Button onClick={props.handleOpen}>Click Me!</Button>
      }>
        <Segment.Group>
          <Segment.Header>
            Danger!
          </Segment.Header>
          <Segment>
            Are you really sure you wish to delete this resource?
          </Segment>
          <Modal.Actions>
            <Button outline kind='secondary' onClick={props.handleClose}>
              Cancel
            </Button>
            <Button kind='danger' onClick={props.handleClose}>
              <Icon name='remove_circle_outline' />Delete
            </Button>
          </Modal.Actions>
        </Segment.Group>
      </Modal>
    )}
  </Modal.Wrapper>
</Playground>

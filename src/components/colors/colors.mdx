---
name: Colors

import { Playground } from 'docz'
import { css } from 'emotion'
import colors from '../../colors'
import Segment from '../segment/segment'

# Colors

<Playground __code=''>
  {Object.entries(colors).map(([key, value]) => {
    const colorCard = css({
      width: 200,
      height: 200,
      float: 'left',
      margin: 8,
      position: 'relative'
    })

    const title = css({
      width: '100%',
      position: 'absolute',
      bottom: 0,
      left: 0,
      textAlign: 'center',
      padding: 10,
      background: 'white'
    })

    return (
      <Segment className={colorCard} style={{ backgroundColor: value }}>
        <div className={title}>
          {key}
        </div>
      </Segment>
    )
  })}
</Playground>

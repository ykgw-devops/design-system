### Nested

<Playground>{
  () => {
    const SubMenu = () => (
      <List as={Segment}>
        <List.Item><label>Some input<Input /></label></List.Item>
        <List.Item>Another submenu item</List.Item>
      </List>)  
      
    const Menu = () => (
      <List as={Segment}>
      <Popup placement='right' content={<SubMenu />}>
          <List.Item>Open submenu</List.Item>
        </Popup>
        <List.Item>Another menu item</List.Item>
      </List>)  
      
    return (
      <Popup content={<Menu />} placement='right'>
        <Button kind='secondary'>Open menu</Button>
      </Popup>) 
    } 
}</Playground>
